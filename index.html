async function signup() { const inputs = document.querySelectorAll("#signupForm input"); const name = inputs[0].value; const email = inputs[1].value; const password = inputs[2].value; try { const user = await auth.createUserWithEmailAndPassword(email, password); await db.collection("users").doc(user.user.uid).set({ name: name, email: email }); alert("Signup successful"); } catch (err) { alert(err.message); } }
async function login() { const inputs = document.querySelectorAll("#loginForm input"); const email = inputs[0].value; const password = inputs[1].value; try { await auth.signInWithEmailAndPassword(email, password); window.location.href = "/home.html"; } catch (err) { alert(err.message); } }
